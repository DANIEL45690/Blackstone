cmake_minimum_required(VERSION 3.10)
project(p2p_crypto)
set(CMAKE_C_STANDARD 99)
set(CMAKE_C_FLAGS "${CMAKE_C_FLAGS} -Wall -O2 -march=native -maes -msse4.2 -mrdrnd")
include_directories(${CMAKE_SOURCE_DIR}/../include)
set(SOURCES ../src/main.c ../src/crypto_init.c ../src/crypto_state.c ../src/sbox_tables.c ../src/gf256_tables.c ../src/permutation_tables.c ../src/encode_decode.c ../src/hash_core.c ../src/signature.c ../src/encryption.c ../src/key_exchange.c ../src/random_gen.c ../src/secure_memory.c ../src/entropy.c ../src/self_test.c)
add_executable(p2p_crypto ${SOURCES})
target_link_libraries(p2p_crypto pthread)
add_library(p2p_crypto_static STATIC ${SOURCES})
set_target_properties(p2p_crypto_static PROPERTIES OUTPUT_NAME p2p_crypto)
enable_testing()
add_executable(test_runner ../tests/test_crypto.c)
target_link_libraries(test_runner p2p_crypto_static pthread)
add_test(NAME CryptoTest COMMAND test_runner)
